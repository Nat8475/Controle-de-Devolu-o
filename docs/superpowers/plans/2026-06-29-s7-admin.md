# S7: Admin (Auditoria + Backup + Configurações) — Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.
>
> **Prerequisite:** S1 (Foundation) and S2 (Database) must be complete. Requires `is_admin = true` on the authenticated user's profile.

**Goal:** Implement three admin-only modules — FormAuditoria (audit log viewer + 10 sub-tabs), FormBackup (manual + scheduled R2 backups), FormConfiguracoes (4-hub settings panel with RBAC, cargo, user, and system configs).

**Architecture:** All admin pages are wrapped in `<RequireAdmin>` (is_admin check from RBACContext). Audit log is read-only, queried from `audit_log` table. Backup triggers a Supabase Edge Function that exports `notas_fiscais` + `transferencias` as JSON to R2. Configurações manages `configs` table rows (key-value) and `cargos`/`cargo_modulos`/`usuario_cargos` for RBAC management.

**Tech Stack:** React 18, TanStack Query v5, Zustand v4, Supabase JS v2, shadcn/ui (Tabs, Dialog, Badge), Lucide React

## Global Constraints

- All S1 constraints apply
- All admin pages: `<RequireAdmin>` component — if `!isAdmin`, redirect to `/acesso-negado`
- Cargo module permissions: list of `Modulo` values managed by checkbox grid
- System configs: stored as rows in `configs` table with `chave` (key) and `valor` (string)
- Backup JSON: `{ exported_at, version, notas_fiscais: [...], transferencias: [...] }`
- `toast()` + `showUndo()` from stores for all mutations
- Date formatting: `toLocaleString('pt-BR')`

```typescript
import { useQuery } from '@tanstack/react-query'
import { supabase } from '@/lib/supabase'
import type { AuditLog } from '@/types/database'

export interface AuditoriaFilters {
  tabela: string
  acao: string
  dataIni: string
  dataFim: string
  page: number
}

const PAGE_SIZE = 50

export function useAuditoria(filters: AuditoriaFilters) {
  return useQuery({
    queryKey: ['auditoria', filters],
    queryFn: async () => {
      let q = supabase
        .from('audit_log')
        .select('*, profiles(nome, email)', { count: 'exact' })
        .order('created_at', { ascending: false })
        .range(filters.page * PAGE_SIZE, (filters.page + 1) * PAGE_SIZE - 1)

      if (filters.tabela) q = q.eq('tabela', filters.tabela)
      if (filters.acao) q = q.eq('acao', filters.acao)
      if (filters.dataIni) q = q.gte('created_at', filters.dataIni)
      if (filters.dataFim) q = q.lte('created_at', filters.dataFim + 'T23:59:59')

      const { data, count, error } = await q
      if (error) throw error
      return { rows: (data ?? []) as Array<AuditLog & { profiles: { nome: string; email: string } | null }>, total: count ?? 0 }
    },
  })
}
```

```typescript
import { serve } from 'https://deno.land/std@0.168.0/http/server.ts'
import { createClient } from 'https://esm.sh/@supabase/supabase-js@2'

const R2_ACCOUNT_ID = Deno.env.get('R2_ACCOUNT_ID')!
const R2_ACCESS_KEY = Deno.env.get('R2_ACCESS_KEY_ID')!
const R2_SECRET_KEY = Deno.env.get('R2_SECRET_ACCESS_KEY')!
const R2_BUCKET = Deno.env.get('R2_BUCKET_NAME')!
const R2_PUBLIC_URL = Deno.env.get('R2_PUBLIC_URL')!

const CORS = { 'Access-Control-Allow-Origin': '*', 'Access-Control-Allow-Headers': 'authorization, content-type' }

serve(async (req) => {
  if (req.method === 'OPTIONS') return new Response(null, { headers: CORS })

  try {
    const authHeader = req.headers.get('Authorization')
    if (!authHeader) return new Response(JSON.stringify({ error: 'Unauthorized' }), { status: 401 })

    // Use service role to bypass RLS for full export
    const supabase = createClient(
      Deno.env.get('SUPABASE_URL')!,
      Deno.env.get('SUPABASE_SERVICE_ROLE_KEY')!
    )

    // Verify caller is admin
    const userClient = createClient(
      Deno.env.get('SUPABASE_URL')!,
      Deno.env.get('SUPABASE_ANON_KEY')!,
      { global: { headers: { Authorization: authHeader } } }
    )
    const { data: { user }, error: authErr } = await userClient.auth.getUser()
    if (authErr || !user) return new Response(JSON.stringify({ error: 'Unauthorized' }), { status: 401 })

    const { data: profile } = await supabase.from('profiles').select('is_admin').eq('id', user.id).single()
    if (!profile?.is_admin) return new Response(JSON.stringify({ error: 'Admin required' }), { status: 403 })

    // Export all tables
    const [notas, transferencias, configs] = await Promise.all([
      supabase.from('notas_fiscais').select('*').order('data', { ascending: false }),
      supabase.from('transferencias').select('*').order('created_at', { ascending: false }),
      supabase.from('configs').select('*'),
    ])

    const exportedAt = new Date().toISOString()
    const payload = JSON.stringify({
      exported_at: exportedAt,
      version: '1.0',
      exported_by: user.email,
      notas_fiscais: notas.data ?? [],
      transferencias: transferencias.data ?? [],
      configs: configs.data ?? [],
    }, null, 2)

    const filename = `backups/backup-${exportedAt.replace(/[:.]/g, '-').slice(0, 19)}.json`

    // Upload to R2 with presigned URL (PUT)
    const body = new TextEncoder().encode(payload)
    const endpoint = `https://${R2_ACCOUNT_ID}.r2.cloudflarestorage.com/${R2_BUCKET}/${filename}`

    // NOTE: R2 write requires AWS Sig V4 — for MVP, use a public-write policy on backups/ prefix
    // or integrate full signing (same pattern as r2-presign function)
    const uploadRes = await fetch(endpoint, {
      method: 'PUT',
      headers: { 'Content-Type': 'application/json', 'Content-Length': String(body.length) },
      body,
    })

    const url = `${R2_PUBLIC_URL}/${filename}`
    const size = Math.round(body.length / 1024)

    // Save backup record to configs table
    await supabase.from('configs').upsert({
      chave: `ultimo_backup`,
      valor: JSON.stringify({ url, filename, exported_at: exportedAt, size_kb: size }),
    }, { onConflict: 'chave' })

    return new Response(JSON.stringify({ url, filename, size_kb: size }), {
      headers: { ...CORS, 'Content-Type': 'application/json' },
    })
  } catch (e) {
    return new Response(JSON.stringify({ error: String(e) }), { status: 500, headers: CORS })
  }
})
```

```typescript
import { useMutation, useQuery, useQueryClient } from '@tanstack/react-query'
import { supabase } from '@/lib/supabase'
import { toast } from '@/stores/toastStore'

export function useUltimoBackup() {
  return useQuery({
    queryKey: ['ultimo-backup'],
    queryFn: async () => {
      const { data } = await supabase.from('configs').select('valor').eq('chave', 'ultimo_backup').single()
      if (!data?.valor) return null
      return JSON.parse(data.valor) as { url: string; filename: string; exported_at: string; size_kb: number }
    },
  })
}

export function useGerarBackup() {
  const qc = useQueryClient()
  return useMutation({
    mutationFn: async () => {
      const { data: session } = await supabase.auth.getSession()
      const res = await fetch(
        `${import.meta.env.VITE_SUPABASE_URL}/functions/v1/backup-export`,
        {
          method: 'POST',
          headers: {
            'Content-Type': 'application/json',
            'Authorization': `Bearer ${session.session?.access_token}`,
          },
        }
      )
      const data = await res.json()
      if (!res.ok) throw new Error(data.error ?? 'Erro ao gerar backup')
      return data as { url: string; filename: string; size_kb: number }
    },
    onSuccess: (data) => {
      qc.invalidateQueries({ queryKey: ['ultimo-backup'] })
      toast(`Backup gerado (${data.size_kb} KB)`, 'ok')
    },
    onError: (e: Error) => toast(e.message, 'err'),
  })
}
```

```typescript
import { useQuery, useMutation, useQueryClient } from '@tanstack/react-query'
import { supabase } from '@/lib/supabase'
import { toast } from '@/stores/toastStore'
import type { Config, Cargo, CargoModulo, Modulo } from '@/types/database'

export function useConfigs() {
  return useQuery({
    queryKey: ['configs'],
    queryFn: async () => {
      const { data, error } = await supabase.from('configs').select('*').order('chave')
      if (error) throw error
      return data as Config[]
    },
  })
}

export function useUpsertConfig() {
  const qc = useQueryClient()
  return useMutation({
    mutationFn: async ({ chave, valor }: { chave: string; valor: string }) => {
      const { error } = await supabase.from('configs').upsert({ chave, valor }, { onConflict: 'chave' })
      if (error) throw error
    },
    onSuccess: () => {
      qc.invalidateQueries({ queryKey: ['configs'] })
      toast('Configuração salva', 'ok')
    },
    onError: (e: Error) => toast(e.message, 'err'),
  })
}

export function useCargos() {
  return useQuery({
    queryKey: ['cargos'],
    queryFn: async () => {
      const { data, error } = await supabase
        .from('cargos')
        .select('*, cargo_modulos(modulo)')
        .order('nome')
      if (error) throw error
      return data as Array<Cargo & { cargo_modulos: { modulo: Modulo }[] }>
    },
  })
}

const ALL_MODULOS: Modulo[] = [
  'notas','lancamento','transferencias','frete','reabertura','venda',
  'email','exportar-pdf','dashboard','relatorios','busca','auditoria','backup','configuracoes'
]

export function useSalvarCargoModulos() {
  const qc = useQueryClient()
  return useMutation({
    mutationFn: async ({ cargoId, modulos }: { cargoId: string; modulos: Modulo[] }) => {
      await supabase.from('cargo_modulos').delete().eq('cargo_id', cargoId)
      if (modulos.length > 0) {
        await supabase.from('cargo_modulos').insert(modulos.map(m => ({ cargo_id: cargoId, modulo: m })))
      }
    },
    onSuccess: () => {
      qc.invalidateQueries({ queryKey: ['cargos'] })
      toast('Permissões atualizadas', 'ok')
    },
    onError: (e: Error) => toast(e.message, 'err'),
  })
}

export function useInsertCargo() {
  const qc = useQueryClient()
  return useMutation({
    mutationFn: async (nome: string) => {
      const { error } = await supabase.from('cargos').insert({ nome })
      if (error) throw error
    },
    onSuccess: () => {
      qc.invalidateQueries({ queryKey: ['cargos'] })
      toast('Cargo criado', 'ok')
    },
    onError: (e: Error) => toast(e.message, 'err'),
  })
}

export { ALL_MODULOS }
```

```typescript
import { useState } from 'react'
import { Plus } from 'lucide-react'
import { Button } from '@/components/ui/button'
import { Input } from '@/components/ui/input'
import { useCargos, useSalvarCargoModulos, useInsertCargo, ALL_MODULOS } from '@/hooks/useConfiguracoes'
import type { Modulo } from '@/types/database'
import { cn } from '@/lib/utils'

const MODULO_LABELS: Record<Modulo, string> = {
  notas: 'Notas', lancamento: 'Lançamento', transferencias: 'Transferências',
  frete: 'Frete', reabertura: 'Reabertura', venda: 'Venda',
  email: 'E-mail', 'exportar-pdf': 'Exportar PDF', dashboard: 'Dashboard',
  relatorios: 'Relatórios', busca: 'Busca', auditoria: 'Auditoria',
  backup: 'Backup', configuracoes: 'Configurações',
}

export function ConfigCargos() {
  const { data: cargos = [] } = useCargos()
  const salvar = useSalvarCargoModulos()
  const inserir = useInsertCargo()
  const [novoCargo, setNovoCargo] = useState('')
  const [editing, setEditing] = useState<Record<string, Set<Modulo>>>({})

  function getModulos(cargoId: string): Set<Modulo> {
    if (editing[cargoId]) return editing[cargoId]
    const cargo = cargos.find(c => c.id === cargoId)
    return new Set(cargo?.cargo_modulos.map(cm => cm.modulo) ?? [])
  }

  function toggleModulo(cargoId: string, modulo: Modulo) {
    const current = new Set(getModulos(cargoId))
    current.has(modulo) ? current.delete(modulo) : current.add(modulo)
    setEditing(e => ({ ...e, [cargoId]: current }))
  }

  function handleSave(cargoId: string) {
    const modulos = Array.from(getModulos(cargoId))
    salvar.mutate({ cargoId, modulos })
    setEditing(e => { const next = { ...e }; delete next[cargoId]; return next })
  }

  return (
    <div className="space-y-6">
      {/* New cargo */}
      <div className="flex gap-2 max-w-sm">
        <Input value={novoCargo} onChange={e => setNovoCargo(e.target.value)}
          placeholder="Nome do novo cargo" onKeyDown={e => e.key === 'Enter' && novoCargo.trim() && (inserir.mutate(novoCargo), setNovoCargo(''))} />
        <Button className="bg-primary" onClick={() => { if (novoCargo.trim()) { inserir.mutate(novoCargo.trim()); setNovoCargo('') } }}>
          <Plus className="w-4 h-4" />
        </Button>
      </div>

      {/* Cargo list */}
      {cargos.map(cargo => {
        const mods = getModulos(cargo.id)
        const hasChanges = !!editing[cargo.id]
        return (
          <div key={cargo.id} className="bg-surface dark:bg-surface-dark rounded-card shadow-soft p-4">
            <div className="flex items-center justify-between mb-3">
              <h4 className="font-semibold text-[var(--text)]">{cargo.nome}</h4>
              <Button size="sm" className="bg-primary" onClick={() => handleSave(cargo.id)} disabled={!hasChanges || salvar.isPending}>
                Salvar permissões
              </Button>
            </div>
            <div className="grid grid-cols-2 sm:grid-cols-4 lg:grid-cols-7 gap-2">
              {ALL_MODULOS.map(m => (
                <label key={m} className={cn(
                  'flex items-center gap-1.5 text-xs cursor-pointer px-2 py-1.5 rounded-btn border transition-colors',
                  mods.has(m) ? 'border-primary bg-primary/10 text-primary' : 'border-[var(--border)] text-[var(--text-muted)]'
                )}>
                  <input type="checkbox" checked={mods.has(m)} onChange={() => toggleModulo(cargo.id, m)} className="w-3 h-3 accent-primary" />
                  {MODULO_LABELS[m]}
                </label>
              ))}
            </div>
          </div>
        )
      })}
    </div>
  )
}
```

```typescript
import { useQuery, useMutation, useQueryClient } from '@tanstack/react-query'
import { supabase } from '@/lib/supabase'
import { toast } from '@/stores/toastStore'
import { useCargos } from '@/hooks/useConfiguracoes'

interface UserWithCargo {
  id: string
  nome: string
  email: string
  is_admin: boolean
  cargo_id: string | null
  cargo_nome: string | null
}

function useUsuarios() {
  return useQuery({
    queryKey: ['usuarios'],
    queryFn: async () => {
      const { data, error } = await supabase
        .from('profiles')
        .select('*, usuario_cargos(cargo_id, cargos(nome))')
        .order('nome')
      if (error) throw error
      return (data ?? []).map((u: any) => ({
        id: u.id,
        nome: u.nome,
        email: u.email,
        is_admin: u.is_admin,
        cargo_id: u.usuario_cargos?.[0]?.cargo_id ?? null,
        cargo_nome: u.usuario_cargos?.[0]?.cargos?.nome ?? null,
      })) as UserWithCargo[]
    },
  })
}

export function ConfigUsuarios() {
  const { data: usuarios = [], isLoading } = useUsuarios()
  const { data: cargos = [] } = useCargos()
  const qc = useQueryClient()

  async function atribuirCargo(userId: string, cargoId: string | null) {
    await supabase.from('usuario_cargos').delete().eq('usuario_id', userId)
    if (cargoId) {
      await supabase.from('usuario_cargos').insert({ usuario_id: userId, cargo_id: cargoId })
    }
    qc.invalidateQueries({ queryKey: ['usuarios'] })
    toast('Cargo atualizado', 'ok')
  }

  async function toggleAdmin(userId: string, current: boolean) {
    await supabase.from('profiles').update({ is_admin: !current }).eq('id', userId)
    qc.invalidateQueries({ queryKey: ['usuarios'] })
    toast(`Admin ${!current ? 'concedido' : 'revogado'}`, 'ok')
  }

  if (isLoading) return <p className="text-sm text-[var(--text-muted)]">Carregando usuários...</p>

  return (
    <div className="overflow-x-auto">
      <table className="w-full border-collapse text-sm">
        <thead>
          <tr className="border-b border-[var(--border)]">
            {['Usuário','E-mail','Cargo','Admin'].map(h => (
              <th key={h} className="px-3 py-2 text-left text-xs font-semibold text-[var(--text-muted)] uppercase">{h}</th>
            ))}
          </tr>
        </thead>
        <tbody>
          {usuarios.map(u => (
            <tr key={u.id} className="border-b border-[var(--border)] hover:bg-[var(--border)]/20">
              <td className="px-3 py-2 font-medium text-[var(--text)]">{u.nome || '—'}</td>
              <td className="px-3 py-2 text-[var(--text-muted)]">{u.email}</td>
              <td className="px-3 py-2">
                <select
                  value={u.cargo_id ?? ''}
                  onChange={e => atribuirCargo(u.id, e.target.value || null)}
                  className="rounded-btn border border-[var(--border)] bg-surface dark:bg-surface-dark text-[var(--text)] text-sm px-2 py-1"
                >
                  <option value="">Sem cargo</option>
                  {cargos.map(c => <option key={c.id} value={c.id}>{c.nome}</option>)}
                </select>
              </td>
              <td className="px-3 py-2">
                <label className="flex items-center gap-2 cursor-pointer">
                  <input
                    type="checkbox"
                    checked={u.is_admin}
                    onChange={() => toggleAdmin(u.id, u.is_admin)}
                    className="w-4 h-4 accent-primary"
                  />
                  <span className="text-xs text-[var(--text-muted)]">{u.is_admin ? 'Sim' : 'Não'}</span>
```
